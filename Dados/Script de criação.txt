CREATE TABLE CLIENTE(
    ID_CLIENTE INTEGER NOT null, 
    RAZAO_SOCIAL VARCHAR(80),
    NOME_FANTASIA VARCHAR(80),
    DOCUMENTO VARCHAR(20),
    TIPO_PESSOA VARCHAR(1),
    NOME VARCHAR(80),
    EMAIL VARCHAR(255),
    SITE VARCHAR(255),
    ENDERECO VARCHAR(255),
    COMPLEMENTO VARCHAR(40),
    NUMERO VARCHAR(20),
    BAIRRO VARCHAR(50),
    IE VARCHAR(30),
    IM VARCHAR(30),
    PRIMARY KEY(ID_CLIENTE)
)
